\newcommand{\foo}{}
\newcommand{\fii}{}

\newcommand{\dma}{
    \arabic{mycount}%

    \StrCount{,\professor,}{,\arabic{mycount},}[\foo]%
    \StrCount{,\alunos,}{,\arabic{mycount},}[\fii]%
    \ifnum\foo>0%
        \cellcolor{green!30}%
    \fi%
    \ifnum\fii>0%
        \cellcolor{yellow!30}%
    \fi%
    \ifnum\numexpr \foo + \fii\relax>1%
        \cellcolor{red!30}%
    \fi%

    \addtocounter{mycount}{\nlinhas}%
    \stepcounter{coluna}%
    \ifnum\value{coluna}>\ncolunas%
        \setcounter{coluna}{1}%
        \stepcounter{linha}%
        \setcounter{mycount}{\value{linha}}%
    \fi%
}
